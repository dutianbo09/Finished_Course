#include<iostream>
using namespace std;
class Engineer
{
public:
	int id;
	char* name;
	char* position;
	Engineer(int pid, char* pn,char* pp)
	{
		id=pid;
		name=new char[strlen(pn)+1];
		if(name!=0)
			strcpy(name,pn);
		strcpy(name,pn);
		position=new char[strlen(pp)+1];
		if(position!=0)
		strcpy(position,pp);		
	}
	Engineer()
	{
	}
	void setEngineer(int pid, char* pn,char* pp)
	{
		id=pid;
		name=new char[strlen(pn)+1];
		if(name!=0)
			strcpy(name,pn);
		position=new char[strlen(pp)+1];
		if(position!=0)
			strcpy(position,pp);		
	}
	void print()
	{
		cout<<id<<endl;
		cout<<name<<endl; 
		cout<<position<<endl;
	}
	~Engineer()
	{
		delete[] name;
		delete[] position;
	}
};
int main()
{
	Engineer e1[4];
	int i;
	for( i=0;i<4;i++)
		e1[i].setEngineer(i+1,"zhangsan","mangner");
	Engineer* pe1=new Engineer(5,"zhangsan","mangner");
	Engineer* pe2=new Engineer(6,"zhangsan","mangner");
	for( i=0;i<4;i++)
		e1[i].print();
	pe1->print();
	pe2->print();
	delete pe1;
	delete pe2;
	return 0;
}
