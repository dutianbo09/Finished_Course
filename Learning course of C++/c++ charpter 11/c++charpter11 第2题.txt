#include<iostream>

int average(const int* array, int size);
double average(const double* array, int size);
int main()
{

	double a[80];
	for(int i=0;i<size;i++)
		{
			cin>>a[i];
		}
	double *array;

double average(const double* array, int size)
{
	double sum=0;
	for(int i=0;i<size;i++)
		sum=sum+*(array+i);
	return sum/size;
}
