#include<iostream>
using namespace std;
const int N=30;
struct student
{
	char num[20];
	char name[20];
	int score[3];
	int total;
	double average;
};
void gettotal(struct student stu[],int N);
void getaverage(struct student stu[],int N);
int main()
{
	int i,j;
	struct student stu[N];
	for(i=0;i<N;i++)
		{
			cin>>stu[i].num;
			cin>>stu[i].name;
			for(j=0;j<3;j++)
				{
					cin>>stu[i].score[j];
				}
		}

	gettotal(stu,N); 
	getaverage(stu,N);
	cout<<endl;
	cout<<endl;
	for(i=0;i<N;i++)
		{
			cout<<stu[i].num<<endl;
			cout<<stu[i].name<<endl;
			cout<<stu[i].total<<endl;
			cout<<stu[i].average<<endl;
		}
	return 0;
}
void gettotal(struct student stu[],int N)
{
	for(int i=0;i<N;i++)
		{
			stu[i].total=stu[i].score[0]+stu[i].score[1]+stu[i].score[2] ;
		}
}
void getaverage(struct student stu[],int N)
{
	for(int i=0;i<N;i++)
		{
			stu[i].average=stu[i].total/3.0;
		}
}
