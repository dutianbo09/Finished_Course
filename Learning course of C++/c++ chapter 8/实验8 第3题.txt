#include <iostream>
using namespace std;
double sum(double a[30][30],int n);
int main ()
{
	int n;
	cin>>n;
	double a[30][30];
	for(int i=0;i<=n-1;i++)
			for(int j=0;j<=n-1;j++)
				cin>>a[i][j];
	cout<<sum(a,n)<<endl;
	return 0;
}
double sum(double a[30][30],int n)
{
	double total=0;
	for(int i=0;i<=n-1;i++)
		total=total+a[i][i];
	return total;
}
